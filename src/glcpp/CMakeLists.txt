set(GLCPP_SRC_FILES
    shader.h
    shader.cpp
    mesh.h
    model.h
    camera.h
    window.cpp
    window.h
    cubemap.cpp
    cubemap.h
    framebuffer.cpp
    framebuffer.h
    )
add_library(glcpp  ${GLCPP_SRC_FILES})

# 컴파일 옵션 설정
target_compile_options(glcpp PRIVATE -std=c++17)

if(WIN32)
  target_include_directories(glcpp PUBLIC ./ ./../../external)
  target_link_directories(glcpp PUBLIC ../../external ../../external/lib)
  target_link_libraries(
    glcpp
    glfw
    glm
    -lpthread
    -lglad
    -lgdi32
    -lopengl32
    -static)
endif(WIN32)

if(APPLE) 
target_include_directories(glcpp PUBLIC ./ ../ ./../../external)
target_link_libraries(glcpp PUBLIC
  STB_IMAGE
  assimp
  imgui
  glm::glm
  glad
  -ldl
  -lpthread)
endif(APPLE)
